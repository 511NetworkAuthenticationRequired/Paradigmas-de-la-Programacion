> [!NOTE]
> Pharo ya tiene `ìsPrime`, que hace lo mismo que las opciones de abajo.

## Opción 1
```st
| z aux esPrimo |

z := (UIManager default request: 'Ingrese un numero.') asInteger.
aux:= 2.
esPrimo:= true.

z > 1
	ifFalse: [^UIManager default alert:  'El numero ingresado no es primo.']
	ifTrue: [ 
		[(esPrimo = true) and: (aux < z)] whileTrue: [ 
			(z \\ aux) = 0
				ifTrue: [ esPrimo:= false ].
			aux:= aux + 1
			]
		].
	esPrimo = true
		ifTrue: [ UIManager default alert: 'El numero ingresado es primo.' ]
		ifFalse: [ UIManager default alert: 'El numero ingresado no es primo.' ].
```

## Opción 2
```st
| n limite divisor flag |

n := (UIManager default request: 'Ingrese un numero') asInteger.

n < 2 ifTrue: [ flag := false. ^ UIManager default alert: 'No es primo' ].
n = 2 ifTrue: [ flag := true. ^ UIManager default alert: 'Es primo' ].
(n even) ifTrue: [ flag := false. ^ UIManager default alert: 'No es primo' ].

limite := n sqrt floor.
divisor := (3 to: limite by: 2) detect: [ :i | n \\ i = 0 ] ifNone: [ nil ].

flag := divisor isNil.

UIManager default alert: (flag ifTrue: [ 'Es primo' ] ifFalse: [ 'No es primo' ]).

```

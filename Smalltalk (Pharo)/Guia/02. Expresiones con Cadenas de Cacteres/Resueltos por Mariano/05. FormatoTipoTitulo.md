# Opción 1
```st
| secuencia secuenciaS |

secuencia := (UIManager default request: 'Ingrese un string:') ifNil: [^self].
secuencia := secuencia asLowercase.
secuenciaS := ''.

1 to: (secuencia size) do: [ :i |
	(i = 1 or: [(secuencia at: (i - 1)) isLetter not] )
		ifTrue: [ secuenciaS := secuenciaS , ((secuencia at: i) asUppercase asString) ]
		ifFalse: [ secuenciaS := secuenciaS, (secuencia at: i) asLowercase asString ]
].

UIManager default alert: secuenciaS.
Transcript show: secuenciaS; cr; flush.
```

# Opción 2
```st
| secuencia secuenciaS |

secuencia := (UIManager default request: 'Ingrese un string:') ifNil: [^self].
secuencia := secuencia asLowercase.
secuenciaS := ''.

secuencia:= secuencia findTokens: Character separators.

Transcript show: secuencia asString.

secuencia do: [ :palabra |
	palabra at: 1 put: ((palabra first) asUppercase).
	secuenciaS:= secuenciaS, ' ', palabra.
].

UIManager default alert: secuenciaS.
```

# Formula
```st
| matriz columnas filas |

columnas:= (UIManager default request: 'Ingrese cantidad de columnas') asInteger.
filas:= (UIManager default request: 'Ingrese cantidad de filas') asInteger.

matriz:= Array new: columnas * filas.

1 to: columnas do: [ :x |
	1 to: filas do: [ :y |
		matriz at: ((x - 1) * columnas + y) put: 0. 
	]
].

Transcript show: matriz asString; cr.
```


# Arreglo de arreglos
```st
| mat fil col x y val |

fil:= 4.
col:= 5.

mat:= Array new: fil.

"Transcript show: m printString; cr; flush."

mat do: [ :i |
	mat at: (mat indexOf: i) put: (Array new: col)
].

x:= (UIManager default request: 'Ingrese la posicion: x') asInteger.
y:= (UIManager default request: 'Ingrese la posicion: y') asInteger.
val:= (UIManager default request: 'Ingrese el valor.') asInteger.

(mat at: x) at: y put: val.


Transcript show: 'Matriz ', fil printString, 'x', col printString; cr.
mat do: [ :f |
    Transcript show: f printString; cr.
].
```

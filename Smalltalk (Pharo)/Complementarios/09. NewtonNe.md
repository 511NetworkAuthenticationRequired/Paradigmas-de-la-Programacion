Calcule el número e utilizando la serie descubierta por Newton:

$$
\sum_{k=0}^{\infty} \frac{1}{k!} \approx e
$$

**Nota:** no utilizar el método factorial existente en Smalltalk.

```st
| numero factorial serie flag |

flag:= false.
[ flag not ] whileTrue: [
    numero := UIManager default request: 'Ingrese un número (o Cancelar para salir)'.
    numero ifNil: [ ^UIManager default inform: 'Operación cancelada.' ].
    (numero isEmpty or: [ numero isAllDigits not ])
        ifTrue: [
            UIManager default alert: 'Debe ingresar un número válido. Intente nuevamente.'
        ]
        ifFalse: [
            numero := numero asNumber.
            flag := true
        ].
].

factorial := [ :n |
	n = 0
		ifTrue: [ 1 ]
		ifFalse: [ n * (factorial value: n - 1) ]
].

serie := [ :n |
	n = 0
		ifTrue: [ 1 ]
		ifFalse: [ (1 / (factorial value: n) + (serie value: (n - 1))) ].
].

UIManager default alert: 'El resultado con n = ', numero printString, ' en la serie del numero e de Newton es: ', (((serie value: numero) asFloat) printShowingDecimalPlaces: 7).
```
